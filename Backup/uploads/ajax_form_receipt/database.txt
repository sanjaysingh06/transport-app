CREATE TABLE receipts (
    id INT AUTO_INCREMENT PRIMARY KEY,
    date DATE,
    transportCode VARCHAR(100),
    transportName VARCHAR(255),
    partyCode VARCHAR(100),
    partyName VARCHAR(255),
    receiptNumber VARCHAR(100),
    lrNumber VARCHAR(100),
    stationFrom VARCHAR(255),
    item VARCHAR(255),
    no_of_parcels INT,
    parcel_rate DECIMAL(10, 2),
    freight DECIMAL(10, 2),
    cartage DECIMAL(10, 2),
    commission DECIMAL(10, 2),
    labour DECIMAL(10, 2),
    other DECIMAL(10, 2),
    total DECIMAL(10, 2),
    refund DECIMAL(10, 2),
    grandTotal DECIMAL(10, 2),
    deliveryDate DATE,
    deliveryPerson VARCHAR(255),
    deliveryChargeByParcel DECIMAL(10, 2),
    deliveryChargeAmount DECIMAL(10, 2)
);
